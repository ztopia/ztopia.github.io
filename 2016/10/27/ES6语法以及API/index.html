ES6语法以及API

发表于 2016-10-27 | 分类于 前端 ， ES6

ES6

let

特点

不允许重复定义

*块 级作用域：在块中使用let定义的变量。在外部无法使用

*没有预解析

暂时性死区：如果在一个作用域中使用了某个变量，在外部无法使用，即使在前面也定义了这个变量

全局变量不再是window对象

eg:

let i=0;
{

console.log(i)  //报错，块级作用域，暂时性死区，变量预解析
let i=2;

}
console.log(i); //注释第一个，值为0，块级作用域

const

特点

定义常量必须赋值，并且不能修改

不允许重复定义

*块 级作用域：在块中使用let定义的变量。在外部无法使用

*没有预解析

暂时性死区：如果在一个作用域中使用了某个变量，在外部无法使用，即使在前面也定义了这个变量

全局变量不再是window对象

类

静态成员和实例成员

有构造函数构造 出来的成员为实例成员

由类创建出来的成员是静态成员 Animal.max_age=10000

本质

类本质上还是一个函数

类中的实例方法都定义在了原型中

类中的静态方法都定义在了自身

// 定义了动物类

   class Animal {
   // 这是构造器
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }

    // 这个方法相当于以前加到原型上的实例方法
    eat() {
        console.log('吃吧');
    }

    // 这个方法相当于以前加到原型上的实例方法
    run() {
        console.log('跑吧');
    }

    // 这个方法加了static修饰符,就是以前加到类身上的静态方法
    static getAnimal(...arg) {
        return new Animal(...arg);
    }
}

var mao = new Animal('猫咪', 5);
console.log(mao.name);
console.log(mao.age);
mao.eat();
mao.run();

// 静态方法由类名直接调用
var bao= Animal.getAnimal('宝宝', 8);
console.log(bao.name);

进阶

如果需要执行多次调用

let p = new Promise(function(yes, no) {
   setTimeout(() => {
        yes();
    }, 2000);
});

// 上一个定时器执行完毕后, 开启下一个定时器
p.then(function() {
    console.log('第一个计时器回调执行，开启第二个');

    // 为了可以在外面调用该实例的then方法,只要return出去就可以了
    return new Promise(function(yes, no) {
        setTimeout(() => {
            yes();
        }, 2000);
    });

})

// 如果then回调中返回的是promise实例,那么我们就可以链式编程得到它
.then(function() {
    console.log('第二个计时器回调执行，开启第三个');

    // 为了可以在外面调用该实例的then方法,只要return出去就可以了
    return new Promise(function(yes, no) {
        setTimeout(() => {
            yes();
        }, 2000);
    });
})

.then(function() {
    console.log('第三个计时器回调执行, 完毕');
});

封装

function getTimeout(time) {
    return new Promise(function(yes, no) {
       setTimeout(() => {
          yes('传哥');
       }, time);
    });
}

let p = getTimeout(5000); // 得到promise实例

p.then(function(data){
    console.log(`第一个计时器回调执行，开启第二个____${data}`);
    return getTimeout(3000);
})

.then(function(data) {
    console.log(`第二个计时器回调执行，开启第三个____${data}`);
    return getTimeout(1000);
})

.then(function() {
    console.log('第三个计时器回调执行, 完毕');
});
